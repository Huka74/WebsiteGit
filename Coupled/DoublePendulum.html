Here we consider a double pendulum consisting of a mass \(m_1\) attached to an anchor point with a massless solid rod of length \(l_1\) and a mass \(m_2\) attached to the first one with a massless solid rod of length \(l_2\). Both are subject to gravity \(g\).
Since both masses are constraint to move on a circle, it is natural to use angles \(\theta_1,\theta_2\) instead of \(x_1,y_1,x_2,y_2\) to describe their motion. The angles are defined as:
This parametrisation is not unique, as the zero of the angles can be chosen arbitrarily. For instance, using \(x=(\sin\theta,-\cos\theta)\) places \(\theta=0\) at the bottom of the circle, but other choices are equally valid and correspond to a rotation of the parametrisation.
Using this convention, we can write the position and velocity of the masses: \begin{equation} \begin{array}{l} \mathbf{x}_1 = l_1\left(\begin{array}{l} \sin\theta_1\\ -\cos\theta_1 \end{array}\right) \qquad & \mathbf{x}_2 = \mathbf{x}_1 + l_2\left(\begin{array}{l} \sin\theta_2\\ -\cos\theta_2 \end{array}\right)\\ \mathbf{v}_1 = l_1\dot{\theta}_1\left(\begin{array}{l} \cos\theta_1\\ \sin\theta_1 \end{array}\right) \qquad & \mathbf{v}_2 = \mathbf{v}_1 + l_2\dot{\theta}_2\left(\begin{array}{l} \cos\theta_2\\ \sin\theta_2 \end{array}\right)\\ \mathbf{v}_1^2 = l_1^2\dot{\theta}_1^2 \qquad & \mathbf{v}_2^2 = l_1^2\dot{\theta}_1^2 + l_2^2\dot{\theta}_2^2 + 2l_1l_2\dot{\theta}_1\dot{\theta}_2\cos(\theta_2-\theta_1) \end{array} \end{equation} With this, the kinetic and potential energies can be computed: \begin{equation} \begin{split} T & = \frac{1}{2}m_1\mathbf{v}_1^2 + \frac{1}{2}m_2\mathbf{v}_2^2\\ & = \frac{1}{2}m_1l_1^2\dot{\theta}_1^2 + \frac{1}{2}m_2\left(l_1^2\dot{\theta}_1^2+l_2^2\dot{\theta}_2^2+2l_1l_2\dot{\theta}_1\dot{\theta}_2\cos(\theta_2-\theta_1)\right)\\ & = \frac{1}{2}(m_1+m_2)l_1^2\dot{\theta}_1^2 + \frac{1}{2}m_2l_2^2\dot{\theta}_2^2 + m_2l_1l_2\dot{\theta}_1\dot{\theta}_2\cos(\theta_2-\theta_1)\\ & \\ V & = m_1gy_1 + m_2gy_2\\ & = -gm_1l_1\cos\theta_1 - gm_2(l_1\cos\theta_1+l_2\cos\theta_2)\\ & = -g(m_1+m_2)l_1\cos\theta_1 - gm_2l_2\cos\theta_2 \end{split} \end{equation} as well as their derivatives: \begin{equation} \begin{split} \frac{\partial V}{\partial \theta_1} & = g(m_1+m_2)l_1\sin\theta_1 & \qquad \frac{\partial V}{\partial \dot{\theta_1}}=0\\ \frac{\partial V}{\partial \theta_2} & = gm_2l_2\sin\theta_2 & \qquad \frac{\partial V}{\partial \dot{\theta_2}}=0\\ \end{split} \end{equation} \begin{equation} \begin{split} \frac{\partial T}{\partial \theta_1} & = m_2l_1l_2\dot{\theta}_1\dot{\theta}_2\sin(\theta_2-\theta_1) & \qquad \frac{\partial T}{\partial \dot{\theta_1}} & = (m_1+m_2)l_1^2\dot{\theta}_1 + m_2l_1l_2\dot{\theta}_2\cos(\theta_2-\theta_1)\\ \frac{\partial T}{\partial \theta_2} & = -m_2l_1l_2\dot{\theta}_1\dot{\theta}_2\sin(\theta_2-\theta_1) & \qquad \frac{\partial T}{\partial \dot{\theta_2}} & = m_2l_2^2\dot{\theta}_2 + m_2l_1l_2\dot{\theta}_1\cos(\theta_2-\theta_1)\\ \end{split} \end{equation} Putting these together to form the Lagrangian \(\mathcal{L}=T-V\), we can compute the relevant quantities for the Euler-Lagrange equations \(\frac{\mathrm{d}\:}{\mathrm{d}t}\frac{\partial \mathcal{L}}{\partial\dot{\theta_i}} - \frac{\partial \mathcal{L}}{\partial \theta_i} = 0\). \begin{equation} \begin{split} \frac{\mathrm{d}\:}{\mathrm{d}t}\frac{\partial \mathcal{L}}{\partial\dot{\theta_1}} & = (m_1+m_2)l_1^2\ddot{\theta}_1 + m_2l_1l_2\ddot{\theta}_2\cos(\theta_2-\theta_1) - m_2l_1l_2\dot{\theta}_2(\dot{\theta}_2-\dot{\theta}_1)\sin(\theta_2-\theta_1) \\ \frac{\mathrm{d}\:}{\mathrm{d}t}\frac{\partial \mathcal{L}}{\partial\dot{\theta_2}} & = m_2l_1l_2\ddot{\theta}_1\cos(\theta_2-\theta_1) + m_2l_2^2\ddot{\theta}_2 - m_2l_1l_2\dot{\theta}_1(\dot{\theta}_2-\dot{\theta}_1)\sin(\theta_2-\theta_1) \\ \frac{\partial \mathcal{L}}{\partial \theta_1} & = m_2l_1l_2\dot{\theta}_1\dot{\theta}_2\sin(\theta_2-\theta_1) - g(m_1+m_2)l_1\sin\theta_1\\ \frac{\partial \mathcal{L}}{\partial \theta_2} & = -m_2l_1l_2\dot{\theta}_1\dot{\theta}_2\sin(\theta_2-\theta_1) - gm_2l_2\sin\theta_2\\ \end{split} \end{equation} The Euler-Lagrange equations for \(\theta_1\) and \(\theta_2\) gives the following system \begin{equation} \begin{split} & \left\{\begin{array}{ll} (m_1+m_2)l_1^2\ddot{\theta}_1 + m_2l_1l_2\ddot{\theta}_2\cos(\theta_2-\theta_1) = m_2l_1l_2\dot{\theta}_2^2\sin(\theta_2-\theta_1) - g(m_1+m_2)l_1\sin\theta_1\\ m_2l_1l_2\ddot{\theta}_1\cos(\theta_2-\theta_1) + m_2l_2^2\ddot{\theta}_2 = -m_2l_1l_2\dot{\theta}_1^2\sin(\theta_2-\theta_1) - gm_2l_2\sin\theta_2\\ \end{array}\right. \\ \\ & \left\{\begin{array}{lll} \ddot{\theta}_1 + \frac{m_2}{m_1+m_2}\frac{l_2}{l_1}\cos(\theta_2-\theta_1)\ddot{\theta}_2 & = \frac{m_2}{m_1+m_2}\frac{l_2}{l_1}\sin(\theta_2-\theta_1)\dot{\theta}_2^2 & - \frac{g}{l_1}\sin\theta_1\\ \frac{l_1}{l_2}\cos(\theta_2-\theta_1)\ddot{\theta}_1 + \ddot{\theta}_2 & = -\frac{l_1}{l_2}\sin(\theta_2-\theta_1)\dot{\theta}_1^2 & - \frac{g}{l_2}\sin\theta_2\\ \end{array}\right. \end{split} \end{equation} which can be written in matrix form \begin{equation} \left(\begin{array}{cc} 1 & \frac{m_2}{m_1+m_2}\frac{l_2}{l_1}\cos(\theta_2-\theta_1) \\ \frac{l_1}{l_2}\cos(\theta_2-\theta_1) & 1 \end{array}\right) \left(\begin{array}{c} \ddot{\theta}_1 \\ \ddot{\theta}_2 \end{array}\right) = \left(\begin{array}{c} \frac{m_2}{m_1+m_2}\frac{l_2}{l_1}\sin(\theta_2-\theta_1)\dot{\theta}_2^2 & - \frac{g}{l_1}\sin\theta_1 \\ -\frac{l_1}{l_2}\sin(\theta_2-\theta_1)\dot{\theta}_1^2 & - \frac{g}{l_2}\sin\theta_2 \end{array}\right) \end{equation} Defining \(m=\frac{m_2}{m_1+m_2}\) and \(l=\frac{l_1}{l_2}\), and solving the system gives the equation of motion for \(\theta_1\) and \(\theta_2\). \begin{equation} \left\{\begin{array}{c} \ddot{\theta}_1 = \frac{1}{1-m\cos^2(\theta_2-\theta_1)}\left(\frac{m}{l}\sin(\theta_2-\theta_1)\left(\dot{\theta}_2^2+l\cos(\theta_2-\theta_1)\dot{\theta}_1^2\right) + \frac{g}{l_1}\left(m\cos(\theta_2-\theta_1)\sin\theta_2 - \sin\theta_1\right) \right)\\ \ddot{\theta}_2 = \frac{1}{1-m\cos^2(\theta_2-\theta_1)}\left(-ml\sin(\theta_2-\theta_1)\left(\dot{\theta}_1^2+l\cos(\theta_2-\theta_1)\dot{\theta}_2^2\right) + \frac{g}{l_2}\left(\cos(\theta_2-\theta_1)\sin\theta_1 - \sin\theta_2\right) \right)\\ \end{array}\right. \end{equation} While these equations cannot be solved analytically, they can be integrated numerically, like in the animation Looking at small angles, we linearize the equations and obtain \begin{equation} \left(\begin{array}{c} \ddot{\theta}_1 \\ \ddot{\theta}_2 \end{array}\right) = \frac{g}{1-m}\left(\begin{array}{cc} -\frac{1}{l_1} & \frac{m}{l_1} \\ \frac{1}{l_2} & -\frac{1}{l_2} \end{array}\right) \left(\begin{array}{c} \theta_1 \\ \theta_2 \end{array}\right) \end{equation} Imposing harmonic solutions : \(\ddot{\mathbf{\theta}}=\lambda \mathbf{\theta}\) We can find the normal modes of the double pendulum by solving the eigenvalue problem \begin{equation} \frac{g}{1-m}\left(\begin{array}{cc} -\frac{1}{l_1} & \frac{m}{l_1} \\ \frac{1}{l_2} & -\frac{1}{l_2} \end{array}\right) \mathbf{\theta} = A\mathbf{\theta} = \lambda\mathbf{\theta} \end{equation} This gives
